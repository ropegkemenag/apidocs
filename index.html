Example

PHP - CURL

<?php

$curl = curl_init();

curl_setopt_array($curl, array(
  CURLOPT_URL => 'https://api.kemenag.go.id/v1/auth/login',
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_ENCODING => '',
  CURLOPT_MAXREDIRS => 10,
  CURLOPT_TIMEOUT => 0,
  CURLOPT_FOLLOWLOCATION => true,
  CURLOPT_HTTP_VERSION => CURL_HTTP_VERSION_1_1,
  CURLOPT_CUSTOMREQUEST => 'POST',
  CURLOPT_POSTFIELDS => array('email' => 'youremail','password' => 'yourpassword'),
  CURLOPT_HTTPHEADER => array(
    'Cookie: cookiesession1=xxxxxx'
  ),
));

$response = curl_exec($curl);

curl_close($curl);
echo $response;

Nodejs - Axios

var axios = require('axios');
var FormData = require('form-data');
var data = new FormData();
data.append('email', 'youremail');
data.append('password', 'yourpassword');

var config = {
  method: 'post',
  url: 'https://api.kemenag.go.id/v1/auth/login',
  headers: {
    'Cookie': 'cookiesession1=xxxxxxx',
    ...data.getHeaders()
  },
  data : data
};

axios(config)
.then(function (response) {
  console.log(JSON.stringify(response.data));
})
.catch(function (error) {
  console.log(error);
});
